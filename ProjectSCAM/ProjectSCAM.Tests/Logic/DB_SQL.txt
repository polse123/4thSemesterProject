DROP TABLE IF EXISTS NodeSets;
DROP TABLE IF EXISTS Nodes;
DROP TABLE IF EXISTS alarms;
DROP TABLE IF EXISTS beers;
DROP TABLE IF EXISTS vibrationvalues;
DROP TABLE IF EXISTS humidityvalues;
DROP TABLE IF EXISTS temperaturevalues;
DROP TABLE IF EXISTS batches;
DROP TABLE IF EXISTS batchqueue;
DROP TABLE IF EXISTS customers;
DROP TABLE IF EXISTS users;
DROP TABLE IF EXISTS machines;
DROP TABLE IF EXISTS recipes;
DROP TABLE IF EXISTS stopreasons;
DROP TABLE IF EXISTS usertypes;

CREATE TABLE usertypes (
	typeid INT PRIMARY KEY,
	role CHAR(20) NOT NULL UNIQUE
);

CREATE TABLE stopreasons (
	stopid INT PRIMARY KEY,
	actionrequired BOOLEAN NOT NULL,
	stopdescription VARCHAR(256) NOT NULL
);

CREATE TABLE recipes (
	beerid INT PRIMARY KEY,
	maxspeed INT NOT NULL,
	name CHAR(20) NOT NULL,
	barley INT NOT NULL,
	hops INT NOT NULL,
	malt INT NOT NULL,
	wheat INT NOT NULL,
	yeast INT NOT NULL
);

CREATE TABLE machines (
	machineid SERIAL PRIMARY KEY,
	ipaddress VARCHAR(40) NOT NULL UNIQUE,
	description VARCHAR(256)
);

CREATE TABLE users (
	userid SERIAL PRIMARY KEY,
	username CHAR(16) NOT NULL UNIQUE,
	password CHAR(32) NOT NULL,
	firstname CHAR(32) NOT NULL,
	lastname CHAR(32) NOT NULL,
	email CHAR(64) NOT NULL UNIQUE,
	phonenumber CHAR(11),
	isactive BOOLEAN NOT NULL,
	usertype INT NOT NULL,
	FOREIGN KEY(usertype) REFERENCES usertypes(typeid) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE customers (
	customerid SERIAL PRIMARY KEY,
	customername CHAR(32) NOT NULL
);

CREATE TABLE batchqueue (
	queueid INT PRIMARY KEY,
	priority INT NOT NULL,
	amount INT NOT NULL,
	speed INT NOT NULL,
	beerid INT NOT NULL,
	FOREIGN KEY(beerid) REFERENCES recipes(beerid) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE batches (
	acceptableproducts INT NOT NULL,
	defectproducts INT NOT NULL,
	timestampstart CHAR(23) NOT NULL,
	timestampend CHAR(23) NOT NULL,
	expirationdate CHAR(10) NOT NULL,
	succeeded BOOLEAN NOT NULL,
	performance CHAR(12) NOT NULL,
	quality CHAR(12) NOT NULL,
	availability CHAR(12) NOT NULL,
	oee CHAR(12) NOT NULL,
	speed INT NOT NULL,
	beerid INT NOT NULL,
	machine INT NOT NULL,
	soldto INT,
	recalled BOOLEAN NOT NULL,
	batchid SERIAL PRIMARY KEY,
	FOREIGN KEY(beerid) REFERENCES recipes(beerid) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(machine) REFERENCES machines(machineid) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(soldto) REFERENCES customers(customerid) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE temperaturevalues (
	value FLOAt NOT NULL,
	timestamp CHAR(23) NOT NULL,
	belongingto INT NOT NULL,
	PRIMARY KEY(timestamp, belongingto),
	FOREIGN KEY(belongingto) REFERENCES batches(batchid) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE humidityvalues (
	value FLOAt NOT NULL,
	timestamp CHAR(23) NOT NULL,
	belongingto INT NOT NULL,
	PRIMARY KEY(timestamp, belongingto),
	FOREIGN KEY(belongingto) REFERENCES batches(batchid) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE vibrationvalues (
	value FLOAt NOT NULL,
	timestamp CHAR(23) NOT NULL,
	belongingto INT NOT NULL,
	PRIMARY KEY(timestamp, belongingto),
	FOREIGN KEY(belongingto) REFERENCES batches(batchid) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE beers (
	productnumber INT NOT NULL,
	acceptable BOOLEAN NOT NULL,
	belongingto INT NOT NULL,
	PRIMARY KEY(productnumber, belongingto),
	FOREIGN KEY(belongingto) REFERENCES batches(batchid) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE alarms (
	alarmid SERIAL PRIMARY KEY,
	timestamp CHAR(23) NOT NULL,
	stopreason INT NOT NULL,
	handledby INT,
	batch INT NOT NULL,
	FOREIGN KEY(stopreason) REFERENCES stopreasons(stopid) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(handledby) REFERENCES users(userid) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(batch) REFERENCES batches(batchid) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE nodes (
	nodeid INT NOT NULL,
	url CHAR(64) NOT NULL,
	namespaceindex INT NOT NULL,
	machine INT NOT NULL,
	PRIMARY KEY(nodeid, machine),
	UNIQUE(url, machine),
	FOREIGN KEY(machine) REFERENCES machines(machineid) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE nodesets (
	machine INT PRIMARY KEY NOT NULL,
	amount INT NOT NULL,
	state INT NOT NULL,
	defect INT NOT NULL,
	acceptable INT NOT NULL,
	amounttoproduce INT NOT NULL,
	machspeed INT NOT NULL,
	temperature INT NOT NULL,
	humidity INT NOT NULL,
	vibration INT NOT NULL,
	stopreason INT NOT NULL,
	batchid INT NOT NULL,
	barley INT NOT NULL,
	hops INT NOT NULL,
	malt INT NOT NULL,
	wheat INT NOT NULL,
	yeast INT NOT NULL,
	maintenanceTrigger INT NOT NULL,
	maintenanceCounter INT NOT NULL,
	FOREIGN KEY(machine) REFERENCES machines(machineid) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(amount, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(state, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(defect, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(acceptable, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(amounttoproduce, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(machspeed, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(temperature, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(humidity, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(vibration, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(stopreason, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(batchid, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(barley, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(hops, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(malt, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(wheat, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(yeast, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(maintenanceTrigger, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY(maintenanceCounter, machine) REFERENCES nodes(nodeid, machine) ON DELETE CASCADE ON UPDATE CASCADE
);
